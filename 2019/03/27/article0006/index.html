1005.数独

1005

题目大意

检验数独的合法性，保证输入为1~9的整数

题解

暴力

代码

#include <cstdio>
#include <cstring> 
int A[10][10];
int cnt[12];
int X[9] = {0, 0, 0, 3, 3, 3, 6, 6, 6}, Y[9] = {0, 3, 6, 0, 3, 6, 0, 3, 6};
int dx[9] = {0, 0, 0, 1, 1, 1, 2, 2, 2}, dy[9] = {0, 1, 2, 0, 1, 2, 0, 1, 2};
bool judge1(int p)
{
    memset(cnt, 0, sizeof(cnt));
    for(int i = 0; i < 9; ++i) ++cnt[A[p][i]];
    for(int i = 1; i <= 9; ++i)
        if(cnt[i] != 1)
            return 1;
    return 0;
}
bool judge2(int p)
{
    memset(cnt, 0, sizeof(cnt));
    for(int i = 0; i < 9; ++i) ++cnt[A[i][p]];
    for(int i = 1; i <= 9; ++i)
        if(cnt[i] != 1)
            return 1;
    return 0;
}
bool judge3(int p)
{
    memset(cnt, 0, sizeof(cnt));
    int xx = X[p], yy = Y[p];
    for(int i = 0; i < 9; ++i) ++cnt[A[xx + dx[i]][yy+ dy[i]]];
    for(int i = 1; i <= 9; ++i)
        if(cnt[i] != 1)
            return 1;
    return 0;
}
void work()
{
    for(int i = 0; i < 9; ++i)
        for(int j = 0; j < 9; ++j)
            scanf("%d", &A[i][j]);
    for(int i = 0; i < 9; ++i)
        if(judge1(i) || judge2(i) || judge3(i))
        {
            printf("Wrong\n");
            return ;
        }
    printf("Right\n");
    return ;
}    
int main()
{
    int n;
    scanf("%d", &n);
    for(int i = 1; i <= n; ++i)
        work();
    return 0;
}

4009.步步为赢

4009

题目大意

明文，密文转换，模拟

题解

模拟

代码

#include <cstdio>
#include <cstring>
char str1[200], str2[200];
int len;
bool check(int p)
{
    int cnt = 0;
    for(int i = 0; i < p; ++i)
        for(int j = i; j < len; j += p)
            if(str1[j] != str2[cnt++])
                return 0;
    return 1;
}
int main()
{
    scanf("%s%s", str1, str2);
    len = strlen(str1);
    if(strlen(str2) != len)
    {
        printf("No Solution\n");
        return 0;
    }
    for(int i = 1; i <= len; ++i)
        if(len % i == 0 && check(i))
        {
            printf("%d\n", i);
            return 0;
        }
    printf("No Solution\n");
    return 0;
}

3016.均分纸牌

3016

题目大意

有n堆纸牌，每一堆都有若干张，且保证总张数为n的整数倍。
将一堆纸牌的部分(或全部)纸牌，转移到相邻的纸牌堆，称之为一次操作。
求解使得每堆纸牌同样多的最少操作数。

题解

直接模拟。很容易发现，对于每一对相邻的纸牌堆，至多需要进行一次操作。
从左往右模拟。多了就分给右边，不足就从右边拿(把右边拿成负数也不妨，反正总数是足够的)。

代码

#include <cstdio>
int A[120], n, summ = 0, cnt = 0;
int main()
{
    scanf("%d", &n);
    for(int i = 1; i <= n; ++i)
    {
        scanf("%d", &A[i]);
        summ += A[i];
    }
    summ /= n;
    for(int i = 1; i < n; ++i)
        if(A[i] != summ)
        {
            A[i + 1] += A[i] - summ;
            A[i] = summ;
            ++cnt;
        }
    printf("%d\n", cnt);
    return 0;
}

4045.日天要读书

4045

题目大意

给定一个1~n的排列，以及m个形如(L,R,X)的询问:
对于每一个询问,在原排列基础上将位置在[L,R]的序列升序排序，排序前后，
若X位置上的数未发生变化，输出Yes,
若X位置上的数发生了变化，输出No.
参见原题目的样例解释

题解

此题应该有简单解法，毕竟是放在入门作业题中的
由于数据范围是$1 \leq n , m\leq 10^4$ 加上实际数据并没有给满(询问的区间长度没给满)，
所以裸跑暴力也勉强能过, 时间复杂度 O(mnlog n)

相当于求静态区间第k小

  1. 莫队+树状数组 $O(m\cdot\sqrt{n}\log{n})$
  2. 整体二分+树状数组 $O(m\cdot\log^2{n})$
  3. 主席树

    代码

    莫队+树状数组

    #include <cstdio>
#include <algorithm>
using namespace std;
#define MAXN 10500
#define lowbit(x) ((x)&-(x))
int tr[MAXN], n;
void _r(int& x)
{
 char c = getchar();
 while(c < '0' || c > '9')
     c = getchar();
 for(x = 0; c >= '0' && c <= '9'; c = getchar())
     x = (x << 3) + (x << 1) + c - '0';
 return ;
}
void modify(int x, int delta)
{
 for(int i = x; i <= n + 10; i += lowbit(i))
     tr[i] += delta;
}
int query(int x)
{
 int ans = 0;
 for(int i = x; i; i -= lowbit(i))
     ans += tr[i];
 return ans;
}
const int I = 100;
struct interval
{
 int L, R, que, id, L2;
 interval(){}
 interval(int _L, int _R, int _que, int _id)
 {
     L = _L;
     R = _R;
     que = _que;
     id = _id;
     L2 = L / I;
 }
}Q[MAXN];
bool cmp(interval a, interval b)
{
 if(a.L2 == b.L2) return a.R < b.R;
 return a.L2 < b.L2;
}
int A[MAXN], m, ans[MAXN];
int main()
{
 _r(n);
 _r(m);
 for(int i = 1; i <= n; ++i) 
     _r(A[i]);
 for(int i = 1, l, r, x; i <= m; ++i)
 {
     _r(l); _r(r); _r(x);
     Q[i] = interval(l, r, x, i);
 }
 sort(Q + 1, Q + m + 1, cmp);
 int X = 1, Y = 0;
 for(int i = 1, L, R; i <= m; ++i)
 {
     L = Q[i].L;
     R = Q[i].R;
     if(Q[i].que < L || Q[i].que > R)
     {
         ans[Q[i].id] = 1;
         continue;
     }
     while(Y < R)
         modify(A[++Y], 1);
     while(Y > R)
         modify(A[Y--], -1);
     while(X < L)
         modify(A[X++], -1);
     while(X > L)
         modify(A[--X], 1);
     ans[Q[i].id] = (query(A[Q[i].que]) == Q[i].que - L + 1);
 }
 for(int i = 1; i <= m; ++i)
     printf(ans[i] ? "Yes\n" : "No\n");
 return 0;
}

    #include <cstdio>
#define MAXN 10500
#define lowbit(x) (x&-x) 
int pos[MAXN], n, m;
int L[MAXN], R[MAXN], key[MAXN], ans[MAXN];
int tmp1[MAXN], tmp2[MAXN], Q[MAXN], _key[MAXN];
int tr[MAXN << 1];
void modify(int x, int delta)
{
 for(; x <= n; x += lowbit(x))
 {
     tr[x] += delta;
 }
 return ;
} 
int query(int x)
{
 int ans = 0;
 if(x == 0)
 {
     return 0;
 }
 for(; x; x -= lowbit(x))
 {
     ans += tr[x];
 }
 return ans;
}
void _r(int& x) {
 char c = getchar();
 while(c < '0' || c > '9') 
 {
     c = getchar();
 }
 for(x = 0; c >= '0' && c <= '9'; c = getchar()) 
 {
     x = (x << 3) + (x << 1) + c - '0';
 }
 return ;
}
void solve(int _L, int _R, int st, int ed)
{
 if(_R == _L)
 {
     for(int i = st; i <= ed; ++i)
     {
         ans[Q[i]] = _L;
     }
     return ;
 }
 int M = (_L + _R) >> 1, c1 = 0, c2 = 0, pp = st;
 for(int i = _L; i <= M; ++i)
 {
     modify(pos[i], 1);
 }
 for(int i = st, I, delta; i <= ed; ++i)
 {
     I = Q[i];
     if((delta = query(R[I]) - query(L[I] - 1)) >= key[I])
     {
         tmp1[c1++] = Q[i];
     }
     else
     {
         tmp2[c2++] = Q[i];
         key[Q[i]] -= delta;
     }
 }
 for(int i = _L; i <= M; ++i)
 {
     modify(pos[i], -1);
 }
 for(int i = 0; i < c1; ++i)
 {
     Q[pp++] = tmp1[i];
 }
 for(int i = 0; i < c2; ++i)
 {
     Q[pp++] = tmp2[i];
 }
 solve(_L, M, st, st+c1-1);
 solve(M+1, _R, st+c1, pp-1);
 return ;
}
int main() 
{
 _r(n);
 _r(m);
 for(int i = 1, x; i <= n; ++i)
 {
     _r(x);
     pos[x] = i;
 }
 for(int i = 1; i <= m; ++i)
 {
     _r(L[i]);
     _r(R[i]);
     _r(_key[i]);
     if(_key[i] >= L[i] && _key[i] <= R[i])
     {
         key[i] = _key[i] - L[i] + 1;
     }
     else
     {
         key[i] = 1;
     }
     Q[i] = i;
 }
 solve(1, n, 1, m);
 for(int i = 1; i <= m; ++i)
 {
     if(_key[i] >= L[i] && _key[i] <= R[i])
     {
         if(pos[ans[i]] == _key[i])
         {
             printf("Yes\n");
         }
         else
         {
             printf("No\n");
         }
     }
     else
     {
         printf("Yes\n");
     }
 }
 return 0;
}

1594.求和

1594

题目大意

此题一语双关
题目背景是割地求和
于是做法也自然是求和

题解

二维前缀和
不过这个题的数据范围有诈，给的不是n，m的范围，而是n*m的范围。
不知道实际数据如何，但保险起见还是动态开辟二维数组吧。

代码

#include <cstdio>
#include <cstdlib>
int n, m, q;
void _r(int& x)
{
    char c = getchar();
    while(c < '0' || c > '9')
        c =getchar();
    for(x = 0; c >= '0' && c <= '9'; c = getchar())
        x = (x << 3) + (x << 1) + c - '0';
    return ;
}
int main()
{
    _r(n);
    _r(m);
    _r(q);
    long long** summ = (long long**) malloc(sizeof(long long*) * (n + 2));
    for(int i = 0; i <= n + 1; ++i)
    {
        summ[i] = (long long*) malloc(sizeof(long long) * (m + 2));
        for(int j = 0; j <= m + 1; ++j)
            summ[i][j] = 0;
    }
    for(int i = 1; i <= n; ++i)
        for(int j = 1, x; j <= m; ++j)
        {
            _r(x);
            summ[i][j] = summ[i-1][j] + summ[i][j-1] - summ[i-1][j-1] + x;
        }
    for(int i = 1, a, b, c, d; i <= q; ++i)
    {
        _r(a); _r(b); _r(c); _r(d);
        printf("%lld\n", summ[c][d] - summ[a-1][d] - summ[c][b-1] + summ[a-1][b-1]);
    }
    return 0;
}

4206. Strict Yan涂色

4206

题目大意

修改部分格子的颜色，使之能够被分成若干k*k的颜色相同的小块。
要求最少化修改操作。

题解

暴力模拟就行了。
因为$$\sum_{k=2}^{+\infty}{\frac{1}{k^2}}=\frac{\pi^2}{6}-1$$
所以效率是O(nm)的

代码

#include <cstdio>
#define MAXN 2050
int cnt[MAXN][MAXN];
int n, m, n2, m2, mx, cc, ans, k2;
char str[2050];
inline int Cnt(int a, int b, int c, int d)
{
    return cnt[c][d] - cnt[c][b-1] - cnt[a-1][d] + cnt[a-1][b-1];
}
int main()
{
    scanf("%d%d", &n, &m);
    ans = n * m;
    mx = (m > n ? m : n);
    n2 = (mx << 1); m2 = (mx << 1);
    for(int i = 1; i <= n; ++i)
    {
        scanf("%s", str + 1);
        for(int j = 1; j <= m; ++j)
            cnt[i][j] = cnt[i-1][j] + cnt[i][j-1] - cnt[i-1][j-1] + str[j] - '0';
        for(int j = m + 1; j <= m2; ++j) cnt[i][j] = cnt[i][j-1];
    }
    for(int i = n + 1; i <= n2; ++i)
        for(int j = 1; j <= m2; ++j)
            cnt[i][j] = cnt[i-1][j];
    for(int k = 2; k <= mx; ++k)
    {
        cc = 0;
        k2 = k * k;
        for(int i = 1; i <= n; i += k)
            for(int j = 1, tmp; j <= m; j += k)
            {
                tmp = Cnt(i, j, i + k - 1, j + k - 1);
                cc += ((tmp << 1) < k2 ? tmp : k2- tmp);
            }
        ans = ans > cc ? cc : ans;
    }
    printf("%d\n", ans);
    return 0;
}
